110 Vector3 position = box->GetPosition();
111 Vector3 scale = box->GetScale();
112
113 if (position.x - scale.x - 0.5f < m_Position.x && m_Position.x < position.x + scale.x + 0.5f &&
114 position.z - scale.z - 0.5f < m_Position.z && m_Position.z < position.z + scale.z + 0.5f)
115 {
116 groundHeight = position.y + scale.y * 2.0f;
117
118 break;
119 }
120 }
121 }
122
123 m_Position = m_GameObject->GetPosition();
