168void Game::LoadpositionToFile(const std::string& filename)
169{
170 std::ifstream inputFile(filename);
171
172 if (inputFile.is_open())
173 {
174 InfoObjData data;
175 int classKind;
176 while (inputFile >> data.pos.x >> data.pos.y >> data.pos.z
177 >> data.scale.x >> data.scale.y >> data.scale.z
178 >> data.rot.x >> data.rot.y >> data.rot.z
179 >> classKind)
180 {
181 switch (classKind)
182 {
183 case CLASS::BOX:
184 {
185 Box* box = AddGameObject<Box>(1);
186 box->SetPosition(data.pos);
187 box->SetRotation(data.rot);
188 box->SetScale(data.scale);
189
190 break;
191 }
192 case CLASS::ENEMY:
193 {
194 Enemy* enemy = AddGameObject<Enemy>(1);
195 enemy->SetPosition(data.pos);
196 enemy->SetRotation(data.rot);
197 enemy->SetScale(data.scale);
198
199 break;
200 }
201 default:
202 break;
203 }
204 }
205
206 inputFile.close();
207 }
208 else
209 {
210 // エラーメッセージを表示またはログに記録
211 }
212}
