27void Audio::Load(const char *FileName)
28{
29
30 // サウンドデータ読込
31 WAVEFORMATEX wfx = { 0 };
32
33 {
34 HMMIO hmmio = NULL;
35 MMIOINFO mmioinfo = { 0 };
36 MMCKINFO riffchunkinfo = { 0 };
37 MMCKINFO datachunkinfo = { 0 };
38 MMCKINFO mmckinfo = { 0 };
39 UINT32 buflen;
40 LONG readlen;
41
42
43 hmmio = mmioOpen((LPSTR)FileName, &mmioinfo, MMIO_READ);
44 assert(hmmio);
45
46 riffchunkinfo.fccType = mmioFOURCC('W', 'A', 'V', 'E');
47 mmioDescend(hmmio, &riffchunkinfo, NULL, MMIO_FINDRIFF);
48
49 mmckinfo.ckid = mmioFOURCC('f', 'm', 't', ' ');
50 mmioDescend(hmmio, &mmckinfo, &riffchunkinfo, MMIO_FINDCHUNK);
51
52 if (mmckinfo.cksize >= sizeof(WAVEFORMATEX))
53 {
54 mmioRead(hmmio, (HPSTR)&wfx, sizeof(wfx));
55 }
56 else
57 {
58 PCMWAVEFORMAT pcmwf = { 0 };
59 mmioRead(hmmio, (HPSTR)&pcmwf, sizeof(pcmwf));
60 memset(&wfx, 0x00, sizeof(wfx));
61 memcpy(&wfx, &pcmwf, sizeof(pcmwf));
62 wfx.cbSize = 0;
63 }
64 mmioAscend(hmmio, &mmckinfo, 0);
65
66 datachunkinfo.ckid = mmioFOURCC('d', 'a', 't', 'a');
67 mmioDescend(hmmio, &datachunkinfo, &riffchunkinfo, MMIO_FINDCHUNK);
68
69
70
71 buflen = datachunkinfo.cksize;
72 m_SoundData = new unsigned char[buflen];
73 readlen = mmioRead(hmmio, (HPSTR)m_SoundData, buflen);
74
75
76 m_Length = readlen;
77 m_PlayLength = readlen / wfx.nBlockAlign;
78
79
80 mmioClose(hmmio, 0);
81 }
82
83
84 // サウンドソース生成
85 m_Xaudio->CreateSourceVoice(&m_SourceVoice, &wfx);
86 assert(m_SourceVoice);
