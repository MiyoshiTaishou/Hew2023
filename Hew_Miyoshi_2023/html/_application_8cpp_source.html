158//-----------------------------------------------------------------------------
159// メインループです.
160//-----------------------------------------------------------------------------
161void Application::MainLoop()
162{
163 MSG msg = {};
164
165 // FPS調整クラス
166 FPS fpsrate(60);
167
168 // ゲームの初期処理
169 Manager::Init(this);
170
171
172 while(WM_QUIT != msg.message)
173 {
174 if (PeekMessage(&msg, nullptr, 0, 0, PM_REMOVE) == TRUE)
175 {
176 TranslateMessage(&msg);
177 DispatchMessage(&msg);
178 }
179 else {
180 uint64_t delta_time = 0;
181
182 // デルタタイムを計算
183 delta_time = fpsrate.CalcDelta();
184
185 Manager::Update(delta_time); // ゲーム更新
186 Manager::Draw(delta_time); // ゲーム描画
187
188 // 規定時間まで時間調整
189 fpsrate.Wait();
190 }
191 }
192
193 // ゲームの終了処理
194 Manager::Uninit();
195
196}
