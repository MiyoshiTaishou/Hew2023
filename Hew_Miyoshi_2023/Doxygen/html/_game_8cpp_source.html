165void Game::LoadpositionToFile(const std::string& filename)
166{
167 std::ifstream inputFile(filename);
168
169 if (inputFile.is_open())
170 {
171 InfoObjData data;
172 int classKind;
173 while (inputFile >> data.pos.x >> data.pos.y >> data.pos.z
174 >> data.scale.x >> data.scale.y >> data.scale.z
175 >> data.rot.x >> data.rot.y >> data.rot.z
176 >> classKind)
177 {
178 switch (classKind)
179 {
180 case CLASS::BOX:
181 {
182 Box* box = AddGameObject<Box>(1);
183 box->SetPosition(data.pos);
184 box->SetRotation(data.rot);
185 box->SetScale(data.scale);
186
187 break;
188 }
189 case CLASS::ENEMY:
190 {
191 Enemy* enemy = AddGameObject<Enemy>(1);
192 enemy->SetPosition(data.pos);
193 enemy->SetRotation(data.rot);
194 enemy->SetScale(data.scale);
195
196 break;
197 }
198 default:
199 break;
200 }
201 }
202
203 inputFile.close();
204 }
205 else
206 {
207 // エラーメッセージを表示またはログに記録
208 }
209}
