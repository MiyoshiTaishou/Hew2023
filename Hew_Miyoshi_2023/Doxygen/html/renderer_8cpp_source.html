499void Renderer::CreateVertexShader( ID3D11VertexShader** VertexShader, ID3D11InputLayout** VertexLayout, const char* FileName )
500{
501
502 FILE* file;
503 long int fsize;
504
505 fopen_s(&file,FileName, "rb");
506 assert(file);
507
508 fsize = _filelength(_fileno(file));
509 unsigned char* buffer = new unsigned char[fsize];
510 fread(buffer, fsize, 1, file);
511 fclose(file);
512
513 m_Device->CreateVertexShader(buffer, fsize, NULL, VertexShader);
514
515
516 D3D11_INPUT_ELEMENT_DESC layout[] =
517 {
518 { "POSITION", 0, DXGI_FORMAT_R32G32B32_FLOAT, 0, 0, D3D11_INPUT_PER_VERTEX_DATA, 0 },
519 { "NORMAL", 0, DXGI_FORMAT_R32G32B32_FLOAT, 0, 4 * 3, D3D11_INPUT_PER_VERTEX_DATA, 0 },
520 { "COLOR", 0, DXGI_FORMAT_R32G32B32A32_FLOAT, 0, 4 * 6, D3D11_INPUT_PER_VERTEX_DATA, 0 },
521 { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, 4 * 10, D3D11_INPUT_PER_VERTEX_DATA, 0 }
522 };
523 UINT numElements = ARRAYSIZE(layout);
524
525 m_Device->CreateInputLayout(layout,
526 numElements,
527 buffer,
528 fsize,
529 VertexLayout);
530
531 delete[] buffer;
532}

554void Renderer::PostProcess()
555{
556 // テクスチャの作成
557 ID3D11Texture2D* m_PostProcessTexture;
558 D3D11_TEXTURE2D_DESC textureDesc{};
559 textureDesc.Width = m_Application->GetWidth();
560 textureDesc.Height = m_Application->GetHeight();
561 textureDesc.MipLevels = 1;
562 textureDesc.ArraySize = 1;
563 textureDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
564 textureDesc.SampleDesc.Count = 1;
565 textureDesc.SampleDesc.Quality = 0;
566 textureDesc.Usage = D3D11_USAGE_DEFAULT;
567 textureDesc.BindFlags = D3D11_BIND_RENDER_TARGET | D3D11_BIND_SHADER_RESOURCE;
568 textureDesc.CPUAccessFlags = 0;
569 textureDesc.MiscFlags = 0;
570 HRESULT hr = Renderer::GetDevice()->CreateTexture2D(&textureDesc, nullptr, &m_PostProcessTexture);
571 if (FAILED(hr))
572 {
573 // エラーハンドリング
574 return;
575 }
576
577 // フレームバッファのレンダーターゲットビューを作成
578 ID3D11RenderTargetView* m_PostProcessRenderTargetView;
579 D3D11_RENDER_TARGET_VIEW_DESC rtvDesc{};
580 rtvDesc.Format = textureDesc.Format;
581 rtvDesc.ViewDimension = D3D11_RTV_DIMENSION_TEXTURE2D;
582 rtvDesc.Texture2D.MipSlice = 0;
583 hr = Renderer::GetDevice()->CreateRenderTargetView(m_PostProcessTexture, &rtvDesc, &m_PostProcessRenderTargetView);
584 if (FAILED(hr))
585 {
586 // エラーハンドリング
587 return;
588 }
589
590 // 頂点データの作成
591 VertexPositionTexture vertices[] =
592 {
593 { Vector3(-1.0f, 1.0f, 0.5f), Vector2(0.0f, 0.0f) }, // 左上
594 { Vector3(1.0f, 1.0f, 0.5f), Vector2(1.0f, 0.0f) }, // 右上
595 { Vector3(-1.0f, -1.0f, 0.5f), Vector2(0.0f, 1.0f) }, // 左下
596 { Vector3(1.0f, -1.0f, 0.5f), Vector2(1.0f, 1.0f) } // 右下
597 };
598
599 // 頂点バッファの作成
600 ID3D11Buffer* m_VertexBuffer;
601 D3D11_BUFFER_DESC vbd{};
602 vbd.Usage = D3D11_USAGE_DEFAULT;
603 vbd.ByteWidth = sizeof(vertices);
604 vbd.BindFlags = D3D11_BIND_VERTEX_BUFFER;
605 vbd.CPUAccessFlags = 0;
606 vbd.MiscFlags = 0;
607 D3D11_SUBRESOURCE_DATA vinitData{};
608 vinitData.pSysMem = vertices;
609 hr = Renderer::GetDevice()->CreateBuffer(&vbd, &vinitData, &m_VertexBuffer);
610 if (FAILED(hr))
611 {
612 // エラーハンドリング
613 return;
614 }
615
616 // インデックスバッファの作成
617 ID3D11Buffer* m_IndexBuffer;
618 uint16_t indices[] =
619 {
620 0, 1, 2,
621 2, 1, 3
622 };
623
624 D3D11_BUFFER_DESC ibd{};
